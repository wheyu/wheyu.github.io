nginx使用

2026-04-14 nginx 技术 运维 后端

简介

Nginx是一个开源的Web服务器，同时Nginx也提供了反向代理和负载均衡的功能。 Nginx通常作为负载均衡器暴露在外网接受用户请求，同时也使用其反向代理的功能，将用户的请求转发到实际提供服务的内网服务器。

官网

  • 官网：http://nginx.org/
  • 官方下载地址：http://nginx.org/en/download.html

配置

# ============================================
# 1. 静态网页服务器
# ============================================
# 适用场景：传统多页面网站，每个URL对应一个真实的HTML文件
# 访问: http://localhost:8080
# 返回: [win] D:/myweb/index.html    [linux] /var/www/myweb/index.html
# 访问: http://localhost:8080/about.html
# 返回: [win] D:/myweb/about.html    [linux] /var/www/myweb/about.html
# 访问: http://localhost:8080/about
# 返回: 404 Not Found（因为不存在about这个文件）
server {
    listen 8080;
    
    location / {
        root D:/myweb;                # [win] root D:/myweb    [linux] root /var/www/myweb
        index index.html;            # 默认首页，访问/时返回index.html
    }
}

# ============================================
# 2. SPA单页应用（Vue/React路由支持）
# ============================================
# 适用场景：Vue/React等前端框架，只有一个HTML文件，前端自己管理路由
# 访问: http://localhost:8080
# 返回: [win] D:/dist/index.html    [linux] /var/www/dist/index.html
# 访问: http://localhost:8080/about
# 返回: [win] D:/dist/index.html（不是about.html，由前端路由显示about页面）
# 访问: http://localhost:8080/user/123
# 返回: [win] D:/dist/index.html（前端路由会解析/user/123并显示对应用户）
server {
    listen 8080;
    
    location / {
        root D:/dist;                 # [win] root D:/dist    [linux] root /var/www/dist
        # $uri = 请求的路径，如/about
        # $uri/ = 把/about当做目录处理
        # /index.html = 上面两个都不存在时，返回index.html
        try_files $uri $uri/ /index.html;  # 关键：找不到文件就返回index.html
    }
}

# ============================================
# 3. 反向代理（路径不变）
# ============================================
# 适用场景：转发API请求到后端，不改变路径
# 访问: http://localhost/api/user
# 转发: http://backend.com/api/user
# 访问: http://localhost/api/user/list
# 转发: http://backend.com/api/user/list
server {
    location /api {
        proxy_pass http://backend.com;        # 路径完整转发，/api保持不变
    }
}

# ============================================
# 4. 反向代理（剥离前缀，最常用）
# ============================================
# 适用场景：前端用/api前缀，后端接口没有这个前缀
# 访问: http://localhost/api/user
# 转发: http://backend.com/user（/api被剥离）
# 访问: http://localhost/api/user/list
# 转发: http://backend.com/user/list
# 
# 关键：proxy_pass末尾的斜杠/会剥离location中的路径
server {
    location /api/ {
        proxy_pass http://backend.com/;       # 末尾斜杠 = 剥离/api
        # 对比：没有斜杠 = 保留/api
    }
}

# ============================================
# 5. 反向代理（添加前缀）
# ============================================
# 适用场景：后端接口有前缀，前端请求没有
# 访问: http://localhost/user
# 转发: http://backend.com/api/user（添加了/api）
# 访问: http://localhost/user/list
# 转发: http://backend.com/api/user/list
server {
    location / {
        # ^/(.*)$ 匹配整个路径，如user/list
        # /api/$1 在前面加上/api，变成/api/user/list
        rewrite ^/(.*)$ /api/$1 break;       # break表示停止重写，直接转发
        proxy_pass http://backend.com;
    }
}

# ============================================
# 6. 跨域解决（前后端分离完整配置）
# ============================================
# 适用场景：前端和后端不同域名，浏览器会拦截跨域请求
# 前端: http://localhost:8080（浏览器访问这个）
# 后端: https://api.example.com（实际API地址）
# 解决方案：让Nginx做中间人，前端请求同源的Nginx，Nginx转发到后端
server {
    listen 8080;
    
    # 前端静态文件
    location / {
        root D:/dist;                         # [win] D:/dist    [linux] /var/www/dist
        try_files $uri $uri/ /index.html;
    }
    
    # 后端API转发
    location /api/ {
        proxy_pass https://api.example.com/;  # 转发到真实后端
        
        # 关键：修改Host，让后端知道请求的是哪个域名
        proxy_set_header Host api.example.com;
        
        # 关键：修改Origin，让后端认为请求来自同源
        proxy_set_header Origin https://api.example.com;
        
        # 可选：传递真实IP
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}

# ============================================
# 7. 301永久重定向
# ============================================
# 适用场景：网站改版，旧URL永久转移到新URL，搜索引擎会更新索引
# 访问: http://localhost/old/page
# 跳转: http://localhost/new/page（浏览器地址栏会变）
# 访问: http://localhost/old/product/123
# 跳转: http://localhost/new/product/123
server {
    location /old {
        # ^/old(.*)$ 匹配/old后面的部分，如/page或/product/123
        # /new$1 拼接成/new/page或/new/product/123
        # permanent 表示301永久重定向
        rewrite ^/old(.*)$ /new$1 permanent;
    }
}

# ============================================
# 8. 302临时重定向
# ============================================
# 适用场景：临时维护、活动页跳转，搜索引擎不会更新索引
# 访问: http://localhost/temp
# 跳转: http://localhost/new-temp（浏览器地址栏会变）
server {
    location /temp {
        return 302 /new-temp;                  # 302临时跳转
    }
}

# ============================================
# 9. HTTP跳转HTTPS
# ============================================
# 适用场景：强制使用HTTPS访问，提高安全性
# 访问: http://example.com
# 跳转: https://example.com
# 访问: http://example.com/about
# 跳转: https://example.com/about
server {
    listen 80;
    server_name example.com;
    
    # $server_name = example.com
    # $request_uri = 原始请求路径，如/about
    return 301 https://$server_name$request_uri;
}

# ============================================
# 10. 负载均衡（多台服务器轮询）
# ============================================
# 适用场景：高并发场景，把请求分发到多台后端服务器
# 访问: http://localhost/api
# 转发: http://192.168.1.10:8080 或 http://192.168.1.11:8080（轮询）
# backend是upstream名称，可以自定义，但是server中proxy_pass必须使用这个名称
# server 192.168.1.10:8080 weight=5; 权重越大优先转发
# server 192.168.1.12:8080 backup;    # 备份服务器（主都挂了才用）
# server 192.168.1.10:8080 max_fails=3 fail_timeout=30s; 配置失败重试3次，失败30秒后重试
# ip_hash(一客户端IP始终访问同一台服务器适用场景：需要保持会话（Session）的应用)
# least_conn(分配给当前连接数最少的服务器适用场景：请求处理时间差异大)
upstream backend {
    # 默认啥没有就是轮询，可设置 ip_hash/least_conn
    server 192.168.1.10:8080;      # 服务器1
    server 192.168.1.11:8080;      # 服务器2
}

server {
    location /api {
        proxy_pass http://backend;          # 轮询转发
    }
}

# ============================================
# 11. 静态资源缓存
# ============================================
# 适用场景：图片、CSS、JS等不常变化的文件，缓存到浏览器
# 访问: http://localhost/css/style.css
# 返回: 文件 + 告诉浏览器缓存30天
# 访问: http://localhost/（HTML）
# 返回: 文件 + 告诉浏览器不缓存（每次都重新请求）
server {
    # 匹配css、js、png等静态文件
    location ~* \.(css|js|png|jpg|svg)$ {
        root D:/dist;                 # [win] D:/dist    [linux] /var/www/dist
        expires 30d;                 # 缓存30天，浏览器再次访问直接用缓存
        add_header Cache-Control "public, immutable";
    }
    
    # HTML文件不缓存
    location / {
        root D:/dist;
        expires -1;                  # -1表示立即过期，每次都重新请求
        add_header Cache-Control "no-cache";
    }
}

# ============================================
# 12. Gzip压缩
# ============================================
# 适用场景：压缩文本类响应，减少传输大小，加快网页加载
# 作用：CSS、JS、JSON等文件压缩后体积减少60-80%
http {
    gzip on;                         # 开启压缩
    gzip_types text/css application/javascript application/json;
    # 只压缩这些类型：CSS、JS、JSON
    gzip_min_length 1024;            # 大于1KB的文件才压缩
    gzip_comp_level 6;               # 压缩级别1-9，6是性能和压缩比的平衡
    gzip_vary on;                    # 添加Vary: Accept-Encoding头
}

# ============================================
# 13. IP白名单限制
# ============================================
# 适用场景：只允许特定IP访问管理后台
# ip是8*4 32位（二进制）  后面/8 表示前8位不变
# 访问: http://localhost/admin
# 允许: 192.168.1.100，其他IP返回403
server {
    location /admin {
        allow 192.168.1.100;         # 允许的IP
        allow 10.0.0.0/8;            # 允许10.x.x.x整个网段
        deny all;                    # 拒绝其他所有IP
        proxy_pass http://backend;
    }
}

# ============================================
# 14. WebSocket代理
# ============================================
# 适用场景：聊天、实时推送等WebSocket连接
# 访问: ws://localhost/ws
# 转发: ws://backend-server:8080
server {
    location /ws {
        proxy_pass http://backend-server:8080;
        proxy_http_version 1.1;               # WebSocket需要HTTP/1.1
        proxy_set_header Upgrade $http_upgrade;    # 升级协议
        proxy_set_header Connection "upgrade";     # 连接升级
        proxy_set_header Host $host;
    }
}
